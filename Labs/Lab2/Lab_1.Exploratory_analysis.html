# The function below converts binary attributes to numeric values (0, 1) and other catigorical attributes in numbers, 
# using one-hot encoding. It's will be useful for us soon
def convert_binary_numericDf():
    numericDf = data.copy(deep=True)
    for column in numericDf.columns:
        if numericDf[column].dtype == "object":
            attr_values_num = len(numericDf[column].value_counts())
            attr_values = list(set(numericDf[column].values.tolist()))
            new_values = {attr_values[i]: i for i in range(len(attr_values))}
            print(new_values)
            
            if attr_values_num == 2:
                for index in numericDf[column].index.tolist():
                    numericDf.loc[index, column] = new_values[numericDf.loc[index, column]]
                numericDf[column] = numericDf[column].apply(int, convert_dtype=True)
            else:
                one_hot_columns = pd.get_dummies(numericDf[column], prefix=column)
                idx = numericDf.columns.get_loc(column)
                numericDf.drop(columns=[column], inplace=True)
                for column_ in one_hot_columns:
                    numericDf.insert(idx, column=column_, value=one_hot_columns[column_])
    return numericDf

# The function below shows outliers in input Data frame, previously converted into numeric data
def show_outliers(data):
    plt.figure(figsize=(15, 10))
    columns_list = [data[column] for column in data.columns if data[column].dtype != "object"]
    columns_names_list = [column for column in data.columns if data[column].dtype != "object"]
    box_plot = plt.boxplot(columns_list)
    fliers_dict = dict()
    for i in range(len(box_plot["fliers"])):
        fliers_data = box_plot["fliers"][i].get_ydata()
        if len(fliers_data) > 0:
            fliers_dict[columns_names_list[i]] = (len(fliers_data), set(fliers_data))
    
    all_outliers = 0
    for key, value in fliers_dict.items():
        print("Attribute: " + '"' + key + '"', " "*(20 - len(key)) + " Number of outliers:",
              value[0], " "*(5 - len(str(value[0]))) + " Outliers: ", value[1])
        all_outliers += value[0]
    print("Number of all outliers: ", all_outliers)
    plt.close()
